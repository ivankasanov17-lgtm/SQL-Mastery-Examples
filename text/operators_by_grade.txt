# Junior (Базовые арифметические и логические операторы)
+    (Сложение)
-    (Вычитание)
*    (Умножение)
/    (Деление)
%    (Остаток от деления / Модуль)
=    (Равно)
<>   (Не равно)
<    (Меньше)
>    (Больше)
<=   (Меньше или равно)
>=   (Больше или равно)
||   (Конкатенация строк / массивов)
~~   (LIKE - Соответствие шаблону)
~~*  (ILIKE - Регистронезависимое соответствие шаблону)

# Middle (Работа с JSON, массивами, регулярными выражениями)
->   (Получить поле JSON / элемент массива)
->>  (Получить поле JSON как текст)
@>   (Содержит - для массивов, JSONB, диапазонов)
<@   (Содержится в - для массивов, JSONB, диапазонов)
&&   (Пересечение - для массивов, диапазонов)
~    (Regex: Соответствует, с учетом регистра)
~*   (Regex: Соответствует, без учета регистра)
!~   (Regex: Не соответствует, с учетом регистра)
!~*  (Regex: Не соответствует, без учета регистра)
!~~  (NOT LIKE)
!~~* (NOT ILIKE)

# Senior (Продвинутая работа с JSON, полнотекстовый поиск, битовые операции)
#>   (Получить JSON объект по пути)
#>>  (Получить JSON объект по пути как текст)
#-   (Удалить ключ/путь из JSONB)
?    (JSONB: Ключ существует)
?&   (JSONB: Все ключи существуют)
?|   (JSONB: Любой из ключей существует)
@@   (Полнотекстовый поиск: tsvector соответствует tsquery)
^    (Возведение в степень)
|/   (Квадратный корень)
||/  (Кубический корень)
&    (Побитовое AND)
|    (Побитовое OR)
#    (Побитовое XOR)
~    (Побитовое NOT - унарный)
<<   (Побитовый сдвиг влево)
>>   (Побитовый сдвиг вправо)

# Специализированные / Геометрические / Сетевые (Редко используются вручную)
# Геометрия (Box, Point, Circle, Polygon и др.)
<<   (Строго слева)
>>   (Строго справа)
&<   (Пересекает слева / не выходит за правую границу)
&>   (Пересекает справа / не выходит за левую границу)
<<|  (Строго ниже)
|>>  (Строго выше)
&<|  (Не выходит за верхнюю границу)
|&>  (Не выходит за нижнюю границу)
<^   (Ниже)
>^   (Выше)
?#   (Пересекается / Intersects)
?-   (Горизонтален?)
?|   (Вертикален?)
?-|  (Перпендикулярен?)
?||  (Параллелен?)
@-@  (Длина или расстояние)
~=   (Геометрическое равенство / совпадение)

# Сетевые адреса (inet/cidr)
<<=  (Содержится в или равно)
>>=  (Содержит или равно)

# Диапазоны (Range types)
-|-  (Смежен с / Adjacent to)
+    (Объединение диапазонов)
*    (Пересечение диапазонов)
-    (Разность диапазонов)
